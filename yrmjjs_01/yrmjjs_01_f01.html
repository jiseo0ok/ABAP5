FORM GET_REQ_DATA .

  SELECT A~REQID,
        A~CARRID,
        C~CARRNAME,
        B~FLDATE,
        A~BOOKID,
        A~MANAGER,
        D~PNAME,
        A~REQDT,
        A~CANCELLED,
        B~CUSTOMID,
        B~CLASS,
        A~MENUNUMBER,
        E~NAME AS CUSTOMID_T,
        F~MAINCOURSE,
        G~MENUNAME AS MAINCOURSE_T
   FROM ZTREQMENU_11 AS A
   INNER JOIN SBOOK AS B ON A~CARRID = B~CARRID AND A~BOOKID = B~BOOKID
   INNER JOIN SCARR AS C ON A~CARRID = C~CARRID
   INNER JOIN ZTEMPLOYEE_11 AS D ON A~MANAGER = D~PERNR
   LEFT JOIN SCUSTOM AS E ON B~CUSTOMID = E~ID
   LEFT JOIN SMENU AS F ON A~MENUNUMBER = F~MENUNUMBER
   INNER JOIN ZTMENU_11 AS G ON A~MENUNUMBER = G~MENUNUMBER
   WHERE A~CARRID IN @S_CAR
     AND B~FLDATE IN @S_FLD
     AND A~REQDT IN @S_RED
     AND A~MANAGER IN @S_MAN
     AND ( ( @P_FIR = 'X' AND B~CLASS = 'F' ) OR
           ( @P_BIZ = 'X' AND B~CLASS = 'C' ) OR
           ( @P_ECO = 'X' AND B~CLASS = 'Y' ) )
   INTO CORRESPONDING FIELDS OF TABLE @GT_LIST.

  PERFORM GET_DOMAIN_CANCELLED.
  PERFORM GET_DOMAIN_CLASS.

ENDFORM.
